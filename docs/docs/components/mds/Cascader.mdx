# Cascader级联选择器

## 基本使用
```tsx

import React from 'react';
import { Cascader } from '@nan-design/react';

interface Option {
  value: string;
  label: string;
  children?: Option[];
}

const options: Option[] = [
  {
    value: 'zhejiang',
    label: 'Zhejiang',
    children: [
      {
        value: 'hangzhou',
        label: 'Hangzhou',
        children: [
          {
            value: 'xihu',
            label: 'West Lake',
          },
        ],
      },
    ],
  },
  {
    value: 'jiangsu',
    label: 'Jiangsu',
    children: [
      {
        value: 'nanjing',
        label: 'Nanjing',
        children: [
          {
            value: 'zhonghuamen',
            label: 'Zhong Hua Men',
          },
        ],
      },
    ],
  },
];

const onChange= (value) => {
};

const App: React.FC = () => (
  <div style={{ height: 100 }}>
    <Cascader 
    options={options} 
    onChange={onChange} 
    placeholder="Please select" />
  </div>
);

export default App;
```

## 移入展开
设置`expandTrigger`属性值为`hover`，鼠标移入展开
`expandTrigger`默认值为`click`，鼠标点击展开
```tsx

interface Option {
  value: string;
  label: string;
  children?: Option[];
}

  {
    value: 'zhejiang',
    label: 'Zhejiang',
    children: [
      {
        value: 'hangzhou',
        label: 'Hangzhou',
        children: [
          {
            value: 'xihu',
            label: 'West Lake',
          },
        ],
      },
    ],
  },
  {
    value: 'jiangsu',
    label: 'Jiangsu',
    children: [
      {
        value: 'nanjing',
        label: 'Nanjing',
        children: [
          {
            value: 'zhonghuamen',
            label: 'Zhong Hua Men',
          },
        ],
      },
    ],
  },
];

const onChange= (value) => {
};

const App: React.FC = () => (
  <div style={{ height: 100 }}>
    <Cascader
    defaultValue={

  </div>
);

export default App;
```

    label: 'Jiangsu',
    children: [
      {

        label: 'Nanjing',
        children: [
          {
            value: 'zhonghuamen',
            label: 'Zhong Hua Men',
          },
        ],
      },
    ],
  },
];

const onChange= (value) => {
};

const App: React.FC = () => (
  <div style={{ height: 100 }}>

## 多选
